def position=(point)
  unless point.respond_to?(:x) && point.respond_to?(:y)
    raise ArgumentError, "expected #{point.inspect}:#{point.class}" +
                          " to respond to #x and #y"
  end

  @bridge.setWindowPosition point.x, point.y
end
