  keys.each do |k|
    yield k, self[k]
  end

def fetch(key, &blk)
  if self.key? key
    return self[key]
  end

  if block_given?
    yield key
  else
    # should be KeyError, but it's 1.9-specific
    raise IndexError, "missing key #{key.inspect}" 
  end
end
