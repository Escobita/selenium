def self.free?(port)
  interfaces = Socket.getaddrinfo("localhost", port).map { |e| e[2] }
  interfaces += ["0.0.0.0", Platform.ip]
  interfaces.each { |address| TCPServer.new(address, port).close }

  true
rescue SocketError, Errno::EADDRINUSE
  false
end
