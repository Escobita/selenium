def rotate(orientation)
  unless ORIENTATIONS.include?(orientation)
    raise ArgumentError, "expected #{ORIENTATIONS.inspect}, got #{orientation.inspect}"
  end

  bridge.setScreenOrientation(orientation.to_s.upcase)
end
