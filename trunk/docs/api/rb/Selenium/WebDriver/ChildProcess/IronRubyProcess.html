def start
  args = @args.dup

  @process                           = System::Diagnostics::Process.new
  @process.StartInfo.UseShellExecute = true
  @process.StartInfo.FileName        = args.shift
  @process.StartInfo.Arguments       = args.join ' '
  @process.start

  self
end
