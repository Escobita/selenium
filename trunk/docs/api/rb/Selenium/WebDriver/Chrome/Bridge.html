def initialize(opts = {})
  @launcher = Launcher.launcher(
    :default_profile => opts[:default_profile],
    :secure_ssl      => opts[:secure_ssl]
  )

  @executor = CommandExecutor.new
  @launcher.launch(@executor.uri)
end
