def screenshot_as(format)
  case format
  when :base64
    bridge.getScreenshot
  when :png
    bridge.getScreenshot.unpack("m")[0]
  else
    raise Error::UnsupportedOperationError, "unsupported format: #{format.inspect}"
  end
end
