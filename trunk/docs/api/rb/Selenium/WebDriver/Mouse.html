def move_to(element, right_by = nil, down_by = nil)
  unless element.kind_of? Element
    raise TypeError, "expected #{Element}, got #{element.inspect}:#{element.class}"
  end

  @bridge.mouseMoveTo element.ref, right_by, down_by
end
